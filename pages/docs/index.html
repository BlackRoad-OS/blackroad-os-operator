// Initialize the BlackRoad client
import { BlackRoad } from '@blackroad/sdk';

const client = new BlackRoad({
  apiKey: process.env.BLACKROAD_API_KEY,
});

// Deploy a worker
const deployment = await client.workers.deploy({
  name: 'my-worker',
  code: workerCode,
  routes: ['api.example.com/*'],
});

console.log(`Deployed to ${deployment.url}`);
